% Appendix A: Embedding-Space Metric Formulations

\label{app:metric-formulations}

\noindent This appendix records the formal definitions behind the active eight-metric structural morphology profile. All quantities are computed on L2-normalized SPECTER2 document embeddings in the original embedding space. For a subfield \(s\), let \(Z_s=\{z_i:i\in s\}\), \(n_s=|Z_s|\), \(c_s=n_s^{-1}\sum_{i\in s}z_i\), and \(r_i=d(z_i,c_s)\), where \(d(\cdot,\cdot)\) is cosine distance.

\medskip
\noindent\textbf{Global dispersion.} The three centroid-based dispersion metrics are
\[
\operatorname{median}_{i\in s}(r_i), \qquad
Q_{0.75}(r)-Q_{0.25}(r), \qquad
Q_{0.90}(r).
\]
They correspond to typical distance from the centroid, middle-spread heterogeneity, and the outer semantic periphery.

\medskip
\noindent\textbf{Local density.} Let \(N_k(i)\) be the directed set of within-subfield nearest neighbors for paper \(i\), with \(k=15\) unless the subfield has fewer available non-self neighbors. Let
\[
D_k(s)=\{d(z_i,z_j): i\in s,\ j\in N_k(i)\},
\qquad
h_j=\sum_{i\in s}\mathbf{1}\{j\in N_k(i)\}.
\]
The local-density metrics are
\[
\operatorname{median}(D_k(s)),
\qquad
\frac{\operatorname{sd}(D_k(s))}{\operatorname{mean}(D_k(s))}.
\]

\medskip
\noindent\textbf{Hubness.}
Hubness is the Gini coefficient of the in-degree counts \(h_j\). If \(h_{(1)}\leq\cdots\leq h_{(n_s)}\), then
\[
G_h(s)=\frac{2\sum_{\ell=1}^{n_s}\ell h_{(\ell)}}{n_s\sum_{\ell=1}^{n_s}h_{(\ell)}}-\frac{n_s+1}{n_s}.
\]

\medskip
\noindent\textbf{Spectral structure.} Let \(\lambda_1,\ldots,\lambda_p\) be the positive PCA explained-variance values for the normalized embeddings of subfield \(s\). The dimensionality proxy and normalized spectral entropy are
\[
D_{80}(s)=\min\left\{m:\frac{\sum_{j=1}^{m}\lambda_j}{\sum_{j=1}^{p}\lambda_j}\geq 0.80\right\},
\qquad
H(s)=-\frac{\sum_{j=1}^{p}p_j\log p_j}{\log p},
\]
where \(p_j=\lambda_j/\sum_{\ell=1}^{p}\lambda_\ell\). The first quantity counts the PCA components needed to explain 80 percent of variance; the second measures how evenly variance is spread across positive-variance directions.

\medskip
\noindent\textbf{Separate semantic-displacement indicator.} Let \(c_s^{early}\) and \(c_s^{late}\) be centroids for the early and late windows. In the full thesis period these are 2000--2004 and 2020--2024. Early--late centroid drift is
\[
\Delta_c(s)=d(c_s^{early},c_s^{late}).
\]
This quantity is reported as net semantic displacement. It is not included in the eight-metric structural profile, profile PCA, static clustering, or morphological distance matrices.
